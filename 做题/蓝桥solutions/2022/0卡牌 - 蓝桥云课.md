#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N=200010;

//二分答案
LL n,m;
int a[N],b[N];
bool check(int x){
    LL v=m;
    for(int i=1;i<=n;++i){
        //直接符合
        if(a[i]>=x) continue;
        //肯定不符合
        if(a[i]+b[i]<x) return false;
        if(a[i]+b[i]>=x&&v>=x-a[i]){
            v-=(x-a[i]);
        }else{
            return false;
        }
    }
    return true;
}
int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;++i) cin>>a[i];
    for(int i=1;i<=n;++i) cin>>b[i];
    int l=0,r=N*2;
    while(l<r){
        int mid=(l+r+1)>>1;
        if(check(mid)) l=mid;
        else r=mid-1;
    }
    cout<<r<<endl;
    return 0;
}
