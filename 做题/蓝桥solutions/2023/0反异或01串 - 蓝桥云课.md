#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<char> s;
    char w;
    while ((w = getchar())!=EOF) {
        s.push_back('0');
        s.push_back(w);
    }
    s.push_back('0');
    int l = 0,r = 0;
    int n = s.size();
    vector<int> m(n,0);
    vector<int> pre(n+1,0);
    if (s[0] == '1') {
        pre[1] = 1;
    }
    int li;
    for (int i = 1; i < n; i++) {
        if (s[i] == '1') {
            pre[i+1] = pre[i] + 1;
        }
        else {
            pre[i+1] = pre[i];
        }
        if (i < r) {
            li = l - (i - l);
            m[i] = min(r-i,m[li]);
        }
        while (i+m[i]+1 < n and i-m[i]-1 >= 0 and s[i+m[i]+1] == s[i-m[i]-1]) {
            m[i] += 1;
        }
        if (i + m[i] > r) {
            l = i;
            r = i + m[i];
        }
    }
    int res = pre[n];
    for (int i=0; i<n; i++) {
        if (s[i] == '1') continue;
        l = i - m[i];
        r = i + m[i];
        res = min(res,pre[l] + (pre[r+1] - pre[l]) / 2 + pre[n] - pre[r+1]);
    }
    cout << res;
}

import java.util.*;

public class Main {

    public static void main(String[] args) {
        List<Character> s = new ArrayList<>();
        Scanner scanner = new Scanner(System.in);
        String input = scanner.nextLine();

        for (char w : input.toCharArray()) {
            s.add('0');
            s.add(w);
        }
        s.add('0');

        int l = 0, r = 0;
        int n = s.size();
        int[] m = new int[n];
        int[] pre = new int[n + 1];

        if (s.get(0) == '1') {
            pre[1] = 1;
        }

        int li;
        for (int i = 1; i < n; i++) {
            if (s.get(i) == '1') {
                pre[i + 1] = pre[i] + 1;
            } else {
                pre[i + 1] = pre[i];
            }

            if (i < r) {
                li = l - (i - l);
                m[i] = Math.min(r - i, m[li]);
            }

            while (i + m[i] + 1 < n && i - m[i] - 1 >= 0 && s.get(i + m[i] + 1) == s.get(i - m[i] - 1)) {
                m[i] += 1;
            }

            if (i + m[i] > r) {
                l = i;
                r = i + m[i];
            }
        }

        int res = pre[n];
        for (int i = 0; i < n; i++) {
            if (s.get(i) == '1') continue;
            l = i - m[i];
            r = i + m[i];
            res = Math.min(res, pre[l] + (pre[r + 1] - pre[l]) / 2 + pre[n] - pre[r + 1]);
        }

        System.out.println(res);
    }
}
