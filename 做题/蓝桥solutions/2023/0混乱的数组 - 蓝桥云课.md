#include <bits/stdc++.h>
using namespace std;
int x, n, a[100005], res, k, r;
int main() {
  scanf("%d", &x);
  n = 1;
  while (n * (n - 1) / 2 < x) n++;
  printf("%d\n", n);
  if (n * (n - 1) / 2 == x) {
    for (int i = 1; i <= n; i++)
      a[i] = n - i + 1;
  } else {
    res = n * (n - 1) / 2 - x;
    if (res <= n / 2) {
      r = n;
      for (int i = 1; i <= res; i++) {
        a[r--] = i;
        a[r--] = i;
      }
      for (int i = r; i >= 1; i--) a[i] = res + (r - i + 1);
    } else {
      res = n - res - 1;
      r = n;
      for (int i = 1; i <= res; i++) {
        a[r--] = i;
        a[r--] = i;
      }
      for (int i = r; i >= 2; i--) a[i] = res + (r - i + 1);
      a[1] = res + 1;
    }
  }
  for (int i = 1; i <= n; i++) printf("%d%c", a[i], i == n? '\n' : ' ');
  return 0;
}
